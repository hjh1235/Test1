调用C#DLL

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Text.RegularExpressions;
using System.Runtime.InteropServices;
using System.Reflection;

namespace MyApplication
{
    public class TestClass
    {
        static void Main(string[] args)
        {
   //加载程序集(dll文件地址)，使用Assembly类   
            Assembly assembly = Assembly.LoadFile(AppDomain.CurrentDomain.BaseDirectory + "WindowsFormsApplication3.dll");

            //获取类型，参数（名称空间+类）   
            Type type = assembly.GetType("WindowsFormsApplication3.Class1");

            //创建该对象的实例，object类型，参数（名称空间+类）   
            object instance = assembly.CreateInstance("WindowsFormsApplication3.Class1");

            //设置Show_Str方法中的参数类型，Type[]类型；如有多个参数可以追加多个   
            Type[] params_type = new Type[2];
            params_type[0] = Type.GetType("System.String");
            params_type[1] = Type.GetType("System.String");
            //设置Show_Str方法中的参数值；如有多个参数可以追加多个   
            Object[] params_obj = new Object[2];
            params_obj[0] = "1";
            params_obj[1] = "2";
            //执行Show_Str方法   
            object value = type.GetMethod("Show_Str", params_type).Invoke(instance, params_obj);
            Console.WriteLine(value.ToString());
            
        }
    }
}
